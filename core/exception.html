Error Handling

Exception Handler is enabled through the configure method by calling the extension method useExceptionHandler of the Appliction builder, which adds the pre-built ExceptionHandlerMiddleware to the request pipeline, and must be in the top order of all other middlewares as shown in code below: 


                  <?php

                  namespace Application;
                  
                  use DevNet\System\Async\Task;
                  use DevNet\Core\Dependency\IServiceCollection;
                  use DevNet\Core\Dispatcher\IApplicationBuilder;
                  use DevNet\Core\Extensions\ServiceCollectionExtensions;
                  use DevNet\Core\Extensions\ApplicationBuilderExtensions;

                  class Startup
                  {
                     ...
                     public function configure(IApplicationBuilder $app)
                     {
                        $app->useExceptionHandler();

                        // middlewars here

                     }
                  }

Custom Error Handling


                  <?php

                  namespace Application;
                  
                  use DevNet\System\Async\Task;
                  use DevNet\Core\Dependency\IServiceCollection;
                  use DevNet\Core\Dispatcher\IApplicationBuilder;
                  use DevNet\Core\Extensions\ServiceCollectionExtensions;
                  use DevNet\Core\Extensions\ApplicationBuilderExtensions;
                  use DevNet\Core\Http\HttpContext;

                  class Startup
                  {
                     ...
                     public function configure(IApplicationBuilder $app)
                     {
                        $app->useExceptionHandler("/error");

                        $app->useRouter();

                        // middlewars here

                        $app->useEndpoint(function($routes)
                        {
                           $routes->mapGet("/error", function(HttpContext $context) : Task
                           {
                              $error = $context->Error;

                              switch ($error->getCode())
                              {
                                 case 404:
                                    $context->Response->Body->write("<h1>404 : Page not found</h1>");
                                    break;
                                 case 403:
                                    $context->Response->Body->write("<h1>403 : Forbidden Access</h1>");
                                    break;
                                 default:
                                    $context->Response->Body->write("<h1>500 : Unexpected Error</h1>");
                                    break;
                              }

                              return Task::completedTask();
                           });
                        }
                     }
                  }
